class Node:
    def __init__(self, data):
        self.data = data 
        self.next = None 
        
class SinglyLinkedList:
    def __init__(self):
        self.Head = None 
    
    def InsertAtEnd(self,data):
        newnode = Node(data)
        if self.Head == None:
            self.Head = newnode
        else:
            temp = self.Head
            while temp.next != None:
                temp = temp.next 
            temp.next = newnode

        newnode = Node(data)
        if self.Head == None:
            self.Head = newnode
        else:
